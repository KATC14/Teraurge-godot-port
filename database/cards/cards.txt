	syntax: type:(type)

		weapon_attack
			
DAMAGE:
	syntax: (damage type)_dmg:(number)
		charisma_cost
		knowledge_cost
		intelligence_cost
		perception_cost
		agility_cost
		strength_cost
		endurance_cost
		hitpoints_cost
	(*special type of damage that isn't reduced by resistances)

COSTS:
	syntax: (damage type)_cost:(number)
		heat_dmg
		cold_dmg
		impact_dmg
		slash_dmg
		pierce_dmg
		magic_dmg
		hitpoints_dmg
		
		charisma_dmg

		intelligence_dmg
		perception_dmg
		agility_dmg
		strength_dmg
		endurance_dmg
		hitpoints_dmg
	(*special type of cost that can kill the player before the target receives damage)

SPECIAL:
	syntax: (special type):(percentage number)
		escape
		evasion
		reflect
		damage_reduction
		interrupt_card
